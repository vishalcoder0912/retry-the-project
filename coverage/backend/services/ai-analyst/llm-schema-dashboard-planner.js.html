function validateDashboardPlan(raw, schemaProfile) {
  const kpis = (raw?.kpis || [])
    .map((item) => sanitizeKpiSpec(item, schemaProfile))
    .slice(0, 8);
 
  const charts = (raw?.charts || raw?.chartSpecs || [])
    .map((item) => sanitizeChartSpec(item, schemaProfile))
    .filter((item) => {
      delete item.data;
      return validateChartCompatibility(item, schemaProfile).ok;
    })
    .slice(0, 10);
 
  return {
    source: `ollama:${DASHBOARD_MODEL}`,
    message: raw?.message || "Dashboard plan generated.",
    kpis,
    charts,
    filters: Array.isArray(raw?.filters) ? raw.filters.slice(0, 8) : [],
    schemaOnly: true,
    model: DASHBOARD_MODEL,
    provider: "ollama",
  };
}
 
function validateCommand(raw, schemaProfile) {
  const action = raw?.action || "ANSWER";
 
  if (raw?.chartSpec && ["GENERATE_CHART", "MODIFY_CHART"].includes(action)) {
    const chartSpec = sanitizeChartSpec(raw.chartSpec, schemaProfile);
    const compatible = validateChartCompatibility(chartSpec, schemaProfile);
    if (!compatible.ok) {
      return {
        action: "ANSWER",
        message: `${compatible.reason} Try a valid chart type for the available schema.`,
        schemaOnly: true,
        provider: "local-validator",
        model: DASHBOARD_MODEL,
      };
    }
 
    return {
      action,
      message: raw?.message || "Chart command generated.",
      chartSpec,
      schemaOnly: true,
      provider: "ollama",
      model: DASHBOARD_MODEL,
    };
  }
 
  if (raw?.kpiSpec && action === "GENERATE_KPI") {
    return {
      action,
      message: raw?.message || "KPI command generated.",
      kpiSpec: sanitizeKpiSpec(raw.kpiSpec, schemaProfile),
      schemaOnly: true,
      provider: "ollama",
      model: DASHBOARD_MODEL,
    };
  }
 
  if (action === "FILTER") {
    return {
      action,
      message: raw?.message || "Filter command generated.",
      filters: Array.isArray(raw?.filters) ? raw.filters : [],
      schemaOnly: true,
      provider: "ollama",
      model: DASHBOARD_MODEL,
    };
  }
 
  if (["DELETE_CHART", "CLEAR_FILTERS"].includes(action)) {
    return {
      action,
      message: raw?.message || "Command generated.",
      schemaOnly: true,
      provider: "ollama",
      model: DASHBOARD_MODEL,
    };
  }
 
  return {
    action: "ANSWER",
    message: raw?.message || "I can help create charts, KPIs, filters, and explain this schema.",
    schemaOnly: true,
    provider: "ollama",
    model: DASHBOARD_MODEL,
  };
}

function localSchemaAnswer(query, schemaProfile) {
  const metric = pickPrimaryMetric(schemaProfile);
  const category = pickPrimaryCategory(schemaProfile);
  const metric2 = pickSecondaryMetric(schemaProfile);
 
  const parts = [
    `${schemaProfile.datasetName} has ${schemaProfile.rowCount} rows and ${schemaProfile.columnCount} columns.`,
    `Detected domain: ${schemaProfile.domain}.`,
  ];
 
  if (metric) parts.push(`Primary metric: ${metric.name} (${metric.role}).`);
  if (category) parts.push(`Primary dimension: ${category.name} (${category.role}).`);
  if (metric && category) parts.push(`A useful chart is average ${metric.name} by ${category.name}.`);
  if (metric && metric2) parts.push(`A useful relationship chart is ${metric2.name} vs ${metric.name}.`);
 
  return parts.join(" ");
}

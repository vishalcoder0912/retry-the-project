function hasColumn(profile, name) {
  if (name === "__row_count__" || name === "count") return true;
  return (profile.columns || []).some((column) => column.name === name);
}
 
function findByName(profile, names = [], roles = []) {
  const normalizedNames = names.map(normalizeColumnName);
 
  return (profile.columns || []).find((column) =>
    normalizedNames.some((name) => column.normalizedName === name)
  ) || (profile.columns || []).find((column) => roles.includes(column.role)) || null;
}
 
function byRole(profile, roles = []) {
  return (profile.columns || []).filter((column) => roles.includes(column.role));
}
 
function uniqueByTitle(items = []) {
  const seen = new Set();
 
  return items.filter((item) => {
    const key = normalizeColumnName(item.title || item.id || JSON.stringify(item));
    if (seen.has(key)) return false;
    seen.add(key);
    return true;
  });
}

function isWorkforceSalary(profile) {
  const names = new Set((profile.columns || []).map((column) => column.normalizedName));
 
  return (
    names.has("salary_usd") &&
    names.has("experience") &&
    names.has("country") &&
    names.has("education")
  );
}

function sanitizeAndFilterKpis(profile, kpis = []) {
  return uniqueByTitle(kpis)
    .map((item) => sanitizeKpiSpec(item, profile))
    .filter((item) => hasColumn(profile, item.metric))
    .slice(0, DEFAULT_MAX_KPIS);
}
 
function sanitizeAndFilterCharts(profile, charts = []) {
  return uniqueByTitle(charts)
    .map((item) => {
      const cleaned = { ...item };
      delete cleaned.data;
      delete cleaned.rows;
      delete cleaned.rawRows;
      return sanitizeChartSpec(cleaned, profile);
    })
    .filter((item) => hasColumn(profile, item.xKey))
    .filter((item) => item.yKey === "count" || hasColumn(profile, item.yKey))
    .slice(0, DEFAULT_MAX_CHARTS);
}
